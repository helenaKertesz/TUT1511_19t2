// Create a node
struct node *new_node(int data) {
    struct node* node = malloc(sizeof(struct node));    

    // malloc will return NULL if it failed
    if( node == NULL){
        fprintf(stderr, "Malloc failed\n");
        exit(1);
    }
    node -> data = data;
    node -> next = NULL;
    return node;
}

// Create and add a node at the start of the list
void insert_at_front(struct overall *o, int data) {
    struct node* node = new_node(data);
    node -> next = o -> start;
    o -> start = node;

    // if the list was previously empty
    // you should made the end point to
    // this new node as well
    if( o -> end == NULL){
        o -> end = node;
    }
}

// Create and add a node before the last element
// if the list is empty simply insert
void insert_before_end(struct overall *o, int data) {
    struct node* node = new_node(data);

    // if you are inserting into an empty list
    // nust make start and end point to the new node
    if( o -> start == NULL ){
        o -> start = node;
        o -> end = node;
    } else {

        struct node* curr = o -> start;

        // keep itterating until curr points to the
        // second last node
        while( curr -> next != o -> end ){
            curr = curr -> next;
        }

        curr -> next = node;
        node -> next = o -> end;
    }
}

struct overall* new_list(){
    struct overall* new_list = malloc(sizeof(struct overall));
    if( new_list == NULL){
        fprintf(stderr, "Malloc failed\n");
        exit(1);
    }
    new_list -> start = NULL;
    new_list -> end = NULL;
    return new_list;
}

void print_list (struct overall *o){
    struct node* curr = o -> start;

    // keep going until curr goes past the
    // end of the list
    while( curr != NULL ){
        printf("%d -> ", curr -> data);
        curr = curr -> next;
    }
    printf("X\n");
}
