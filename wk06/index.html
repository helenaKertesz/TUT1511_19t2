ASCII and getchar()

In C, characters are stored as numbers. We can use getchar() to read in one character at a time and putchar() to print out one character.

sum_digits.c download 

#include <stdio.h>

int main(){ 

    int digit_sum = 0;
    int digit_count = 0;

    // read in characters
    int c = getchar();

    while( c != EOF ){
        if( c >= '0' && c <= '9'){
            digit_sum = digit_sum + ( c - '0' );
            digit_count++;
        }
        c = getchar();
    }

    printf("Input contained %d digits which summed to %d\n",
            digit_count, digit_sum);

    return 0;

}

#include <stdio.h>

int main( int argc, char* argv[] ){

    int arg_count = 1; // start at 2nd argument
    int char_count = 0;

    while( arg_count < argc ){
        // if I wanted to print out the entire argument
        // printf("%s\n", argv[arg_count]);

        char_count = 0;
        while( argv[arg_count][char_count] != '\0'){
            if(argv[arg_count][char_count] >= '0' &&
                    argv[arg_count][char_count] <= '9'){

                printf("%c\n", argv[arg_count][char_count]);

            }
            char_count++;
        }

        arg_count++;
    }

}
